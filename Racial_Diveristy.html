Now, lets find out what it would look like if UT was proportional to the state demographics (roughly). To do this, we need to calculate the percentage of each race State.Percent

# Total number of residents in the state
State.Total = sum(demographics$Texas, na.rm=T)

# States racial breakdown as a percentage of population
State.Percent = demographics$Texas / State.Total # Rough breakdown, not accounting for everyone

# Save vector to data frame
demographics$State.Percent <- State.Percent

And multiply that by the total number of students at UT

# Total UT students
UT.Total = sum(demographics$Utexas, na.rm=T)

# Find 'ideal' numbers if UT reflected state demographics
UT.by.PercentState = State.Percent * UT.Total

# Bind to data frame
demographics$UT.by.PercentState <- UT.by.PercentState

Plot

require(ggplot2)
require(ggthemes)

ggplot(demographics, aes(x=Race)) + 
  geom_bar(aes(y=UT.by.PercentState), stat = "identity", alpha=I(.8), fill=I("black")) + 
  geom_bar(aes(y=Utexas), stat = "identity", alpha=I(.3), fill=I("green"), color=I("#BF5700")) + 
  labs(y="Number of students at UT") + theme_economist_white() + coord_flip() + 
  ggtitle("Representation of UT Students \nvs Texas Population by race")
